-- TOTAL CUSTOMERS
SELECT COUNT(*) AS TotalCustomers
FROM telco_customer_churn;

-- CUSTOMER STATUS BREAKDOWN
SELECT `Customer Status`, Count(*) as TotalCustomers
FROM telco_customer_churn_status
Group by `Customer Status`;

-- CHURN RATE
Select Round( 
Sum( CASE WHEN `Customer Status` = "Churned" then 1 else 0 end) *100.0 / Count(*),2)
as Churn_rate 
from telco_customer_churn_status;

-- CHURN BY GENDER
SELECT 
d.Gender,
COUNT(*) AS TotalCustomers,
SUM(CASE WHEN st.`Customer Status` = 'Churned' THEN 1 ELSE 0 END) AS ChurnedCustomers
FROM telco_customer_churn_demographics d
JOIN telco_customer_churn_status st
ON d.CustomerID = st.CustomerID
GROUP BY d.Gender;


-- CHURN BY INTERNET TYPE
Select s.`Internet Type`, 
count(*) as Totalcustomers,
sum( case when `Customer Status` = 'Churned' then 1 else 0 end) as ChrunedCustomers
From telco_customer_churn_status st
join telco_customer_churn_services s
on s.CustomerID = st.CustomerID
Group by s.`Internet Type`;

-- AGE GROUP ANALYSIS

SELECT
CASE
WHEN d.Age < 30 THEN 'Under 30'
WHEN d.Age BETWEEN 30 AND 50 THEN '30-50'
ELSE 'Above 50' END AS AgeGroup,
COUNT(*) AS TotalCustomers,
SUM(CASE WHEN st.`Customer Status` ='Churned' THEN 1 ELSE 0 END) AS ChurnedCustomers
FROM telco_customer_churn_demographics d
JOIN telco_customer_churn_status st
ON d.CustomerID = st.CustomerID
GROUP BY AgeGroup;

Select
